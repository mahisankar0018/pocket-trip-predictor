import java.util.*; /* * Trip Cost Estimator * Console-based Java Application * Works perfectly in VS Code */ interface CostComponent { double calculateCost(); } class TravelCost implements CostComponent { private double distance; private double costPerKm; public TravelCost(double distance, double costPerKm) { this.distance = distance; this.costPerKm = costPerKm; } @Override public double calculateCost() { return distance * costPerKm; } } class StayCost implements CostComponent { private int days; private double costPerDay; public StayCost(int days, double costPerDay) { this.days = days; this.costPerDay = costPerDay; } @Override public double calculateCost() { return days * costPerDay; } } class FoodCost implements CostComponent { private int days; private double costPerDay; public FoodCost(int days, double costPerDay) { this.days = days; this.costPerDay = costPerDay; } @Override public double calculateCost() { return days * costPerDay; } } class MiscCost implements CostComponent { private double[] expenses; public MiscCost(double[] expenses) { this.expenses = expenses; } @Override public double calculateCost() { double sum = 0; for (double e : expenses) { sum += e; } return sum; } } public class TripCostEstimator { public static void main(String[] args) { Scanner sc = new Scanner(System.in); List components = new ArrayList<>(); try { System.out.println("==================================="); System.out.println(" TRIP COST ESTIMATOR"); System.out.println("==================================="); System.out.print("Enter Origin City: "); String origin = sc.nextLine(); System.out.print("Enter Destination City: "); String destination = sc.nextLine(); System.out.print("Enter Distance (in KM): "); double distance = sc.nextDouble(); System.out.print("Enter Travel Cost per KM: "); double costPerKm = sc.nextDouble(); components.add(new TravelCost(distance, costPerKm)); System.out.print("Enter Number of Days: "); int days = sc.nextInt(); System.out.print("Enter Stay Cost per Day: "); double stayPerDay = sc.nextDouble(); components.add(new StayCost(days, stayPerDay)); System.out.print("Enter Food Cost per Day: "); double foodPerDay = sc.nextDouble(); components.add(new FoodCost(days, foodPerDay)); System.out.print("Enter Number of Miscellaneous Expenses: "); int n = sc.nextInt(); double[] misc = new double[n]; for (int i = 0; i < n; i++) { System.out.print("Expense " + (i + 1) + ": "); misc[i] = sc.nextDouble(); } components.add(new MiscCost(misc)); double total = 0; for (CostComponent c : components) { total += c.calculateCost(); } System.out.println("\n========== TRIP SUMMARY =========="); System.out.println("From : " + origin); System.out.println("To : " + destination); System.out.println("Days : " + days); System.out.println("---------------------------------"); System.out.println("Total Estimated Cost: ₹" + total); System.out.println("================================="); } catch (InputMismatchException e) { System.out.println("❌ Invalid input! Please enter numeric values only."); } finally { sc.close(); } } }
