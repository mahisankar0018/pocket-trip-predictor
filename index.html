import javax.swing.*; import java.awt.*; import java.awt.event.*; public class TripCostEstimator extends JFrame { JTextField originField, destinationField, distanceField, daysField; JTextField fuelField, hotelField, foodField; JComboBox modeBox; JTextArea resultArea; public TripCostEstimator() { setTitle("Trip Cost Estimator"); setSize(500, 600); setDefaultCloseOperation(EXIT_ON_CLOSE); setLocationRelativeTo(null); setLayout(new BorderLayout()); JPanel form = new JPanel(new GridLayout(9, 2, 10, 10)); form.setBorder(BorderFactory.createEmptyBorder(15, 15, 15, 15)); originField = new JTextField(); destinationField = new JTextField(); distanceField = new JTextField(); daysField = new JTextField(); fuelField = new JTextField(); hotelField = new JTextField(); foodField = new JTextField(); modeBox = new JComboBox<>(new String[]{"Car", "Train", "Flight"}); form.add(new JLabel("Origin City:")); form.add(originField); form.add(new JLabel("Destination City:")); form.add(destinationField); form.add(new JLabel("Distance (km):")); form.add(distanceField); form.add(new JLabel("Number of Days:")); form.add(daysField); form.add(new JLabel("Travel Mode:")); form.add(modeBox); form.add(new JLabel("Fuel Price (per litre):")); form.add(fuelField); form.add(new JLabel("Hotel Cost (per night):")); form.add(hotelField); form.add(new JLabel("Food Cost (per day):")); form.add(foodField); JButton calculateBtn = new JButton("Calculate Trip Cost"); resultArea = new JTextArea(8, 30); resultArea.setEditable(false); resultArea.setFont(new Font("Monospaced", Font.PLAIN, 13)); calculateBtn.addActionListener(e -> calculateCost()); add(form, BorderLayout.NORTH); add(calculateBtn, BorderLayout.CENTER); add(new JScrollPane(resultArea), BorderLayout.SOUTH); } private void calculateCost() { try { double distance = Double.parseDouble(distanceField.getText()); int days = Integer.parseInt(daysField.getText()); double fuelPrice = Double.parseDouble(fuelField.getText()); double hotel = Double.parseDouble(hotelField.getText()); double food = Double.parseDouble(foodField.getText()); String mode = (String) modeBox.getSelectedItem(); double travelCost = 0; if (mode.equals("Car")) { travelCost = (distance / 15) * fuelPrice; } else if (mode.equals("Train")) { travelCost = distance * 1.5; } else if (mode.equals("Flight")) { travelCost = distance * 5; } double hotelCost = hotel * days; double foodCost = food * days; double total = travelCost + hotelCost + foodCost; resultArea.setText( "Trip Cost Estimation\n" + "---------------------------\n" + "From: " + originField.getText() + "\n" + "To: " + destinationField.getText() + "\n\n" + "Travel Cost : ₹ " + String.format("%.2f", travelCost) + "\n" + "Hotel Cost : ₹ " + String.format("%.2f", hotelCost) + "\n" + "Food Cost : ₹ " + String.format("%.2f", foodCost) + "\n" + "---------------------------\n" + "Total Cost : ₹ " + String.format("%.2f", total) ); } catch (Exception ex) { JOptionPane.showMessageDialog(this, "Please enter valid numeric values", "Input Error", JOptionPane.ERROR_MESSAGE); } } public static void main(String[] args) { SwingUtilities.invokeLater(() -> { new TripCostEstimator().setVisible(true); }); } }
