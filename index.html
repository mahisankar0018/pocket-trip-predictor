import java.util.*; interface CostComponent { double calculate(); } class FuelCost implements CostComponent { private double distanceKm; private double fuelEfficiencyKmpl; private double fuelPricePerLitre; public FuelCost(double distanceKm, double fuelEfficiencyKmpl, double fuelPricePerLitre) { this.distanceKm = distanceKm; this.fuelEfficiencyKmpl = fuelEfficiencyKmpl; this.fuelPricePerLitre = fuelPricePerLitre; } @Override public double calculate() { return (distanceKm / fuelEfficiencyKmpl) * fuelPricePerLitre; } } class AdditionalCosts implements CostComponent { private double[] costs; public AdditionalCosts(double[] costs) { this.costs = costs; } @Override public double calculate() { double total = 0; for (double cost : costs) { total += cost; } return total; } } class TripCostEstimator { private List components = new ArrayList<>(); public void addComponent(CostComponent component) { components.add(component); } public double estimateTotalCost() { double total = 0; for (CostComponent component : components) { total += component.calculate(); } return total; } public static void main(String[] args) { try { Scanner scanner = new Scanner(System.in); System.out.print("Enter distance (km): "); double distance = Double.parseDouble(scanner.nextLine()); System.out.print("Enter fuel efficiency (km/l): "); double efficiency = Double.parseDouble(scanner.nextLine()); System.out.print("Enter fuel price (per litre): "); double fuelPrice = Double.parseDouble(scanner.nextLine()); System.out.print("Enter lodging cost: "); double lodging = Double.parseDouble(scanner.nextLine()); System.out.print("Enter food cost: "); double food = Double.parseDouble(scanner.nextLine()); System.out.print("Enter miscellaneous cost: "); double misc = Double.parseDouble(scanner.nextLine()); TripCostEstimator estimator = new TripCostEstimator(); estimator.addComponent(new FuelCost(distance, efficiency, fuelPrice)); estimator.addComponent(new AdditionalCosts(new double[]{lodging, food, misc})); double totalCost = estimator.estimateTotalCost(); System.out.printf("Estimated Trip Cost: â‚¹%.2f%n", totalCost); } catch (NumberFormatException e) { System.out.println("Invalid input. Please enter numeric values."); } catch (Exception e) { System.out.println("An unexpected error occurred: " + e.getMessage()); } } }
